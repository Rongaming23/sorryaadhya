<script>

function filterCases(){

let value =
document.getElementById("searchBox")
.value
.toLowerCase();

let cards =
document.querySelectorAll("#caseList .card");

cards.forEach(card=>{

let txt =
card.innerText.toLowerCase();

card.style.display =
txt.includes(value)
? "block"
: "none";

});

}
