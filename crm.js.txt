import { db, auth } from "./firebase.js";

import {
collection,
addDoc,
onSnapshot,
deleteDoc,
doc,
query,
orderBy
} from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";

import {
signOut,
onAuthStateChanged
} from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js";

const casesRef = collection(db,"cases");

window.logout = async function(){
await signOut(auth);
window.location.href="login.html";
};

onAuthStateChanged(auth,(user)=>{

if(!user){
window.location.href="login.html";
return;
}

loadCases();

});

window.saveCase = async function(){

const ticketId =
"WOW-" +
new Date().toISOString().slice(0,10).replaceAll("-","") +
"-" +
Date.now().toString().slice(-5);

const caseData = {

ticketId,

name: document.getElementById("customerName").value,

phone: document.getElementById("phone").value,

orderId: document.getElementById("orderId").value,

orderDate: document.getElementById("orderDate").value,

outlet: document.getElementById("outlet").value,

deliveryPartner:
document.getElementById("deliveryPartner").value,

issue:
document.getElementById("issueType").value,

status:
document.getElementById("status").value,

concern:
document.getElementById("customerConcern").value,

agent:
auth.currentUser.email,

createdAt:
new Date().toISOString()

};

await addDoc(casesRef,caseData);

alert("Case Saved Successfully");

document.getElementById("customerName").value="";
document.getElementById("phone").value="";
document.getElementById("orderId").value="";
document.getElementById("orderDate").value="";
document.getElementById("outlet").value="";
document.getElementById("customerConcern").value="";

};

function loadCases(){

const q = query(
casesRef,
orderBy("createdAt","desc")
);

onSnapshot(q,(snapshot)=>{

let html="";

let total=0;
let open=0;
let resolved=0;
let escalated=0;

snapshot.forEach((docSnap)=>{

const data = docSnap.data();

total++;

if(data.status==="Open") open++;
if(data.status==="Resolved") resolved++;
if(data.status==="Escalated") escalated++;

html += `

<div class="card" style="margin-bottom:10px;">

<b>${data.ticketId}</b>

<br><br>

Customer : ${data.name}<br>
Phone : ${data.phone}<br>
Order ID : ${data.orderId}<br>
Outlet : ${data.outlet}<br>
Issue : ${data.issue}<br>
Status : ${data.status}<br>
Partner : ${data.deliveryPartner}<br>
Agent : ${data.agent}<br>

<br>

<button
class="danger"
onclick="deleteCase('${docSnap.id}')"
>
Delete
</button>

</div>

`;

});

document.getElementById("caseList").innerHTML =
html || "No Cases Found";

document.getElementById("totalCases").innerText=total;
document.getElementById("openCases").innerText=open;
document.getElementById("resolvedCases").innerText=resolved;
document.getElementById("escalatedCases").innerText=escalated;

});

}

window.deleteCase = async function(id){

if(!confirm("Delete this case?")) return;

await deleteDoc(doc(db,"cases",id));

};
